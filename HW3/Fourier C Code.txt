#include "MK22F51212.h"
#include "MCG.h"
#include "TimerInt.h"
#include "DAC.h"
#include "yk_tables.h"
#include <stdint.h>

// Set the first sample to be read
volatile uint16_t idx = 0;

// Table selection upon start up
volatile uint16_t curK = 1;

//Points to one of the tables defined in yk_tables.h
volatile const uint16_t *table = 0;

// Configures SW2 (PTC1) and SW3 (PTB17) as GPIO inputs w/ internal pull-up resistor
void Switches_Init(void){
    // Enable clock gate to PORTB and PORTC modules
    SIM->SCGC5 |= SIM_SCGC5_PORTB_MASK | SIM_SCGC5_PORTC_MASK;

    // SW2
    PORTC->PCR[1]  = PORT_PCR_MUX(1) | PORT_PCR_PE_MASK | PORT_PCR_PS_MASK;

    // SW3
    PORTB->PCR[17] = PORT_PCR_MUX(1) | PORT_PCR_PE_MASK | PORT_PCR_PS_MASK;
}

// LED Initialization
// Configures PTD5 as GPIO output
void LED_Init(void){
    // Enable clock to PORTD
    SIM->SCGC5 |= SIM_SCGC5_PORTD_MASK;

    // Configure PTD5 as GPIO
    PORTD->PCR[5] = PORT_PCR_MUX(1);

    // Set PTD5 as output
    PTD->PDDR |= (1u << 5);

    // Turn LED OFF initially 
    PTD->PSOR = (1u << 5);
}

void LED_On(void)  { 
    // Drives LED pin low (led ON)
    PTD->PCOR = (1u << 5); 
}

void LED_Off(void) { 
    // Drives LED pin high (led OFF)
    PTD->PSOR = (1u << 5); 
}

uint8_t SW2Pressed(void){
    // Checks if input is low, indicates the switch is pressed
    return ((PTC->PDIR & (1u << 1)) == 0u);
}

uint8_t SW3Pressed(void){
		// Checks if input is low, indicates the switch is pressed
    return ((PTB->PDIR & (1u << 17)) == 0u);
}

// Prevents the button from being read numerous times once pressed
void Debounce(void){
    for (volatile uint32_t d = 0; d < 50000u; d++) {
        __NOP();   
    }
}

void PIT0_IRQHandler(void){

    LED_On();  

    NVIC_ClearPendingIRQ(PIT0_IRQn);	
    PIT->CHANNEL[0].TFLG = PIT_TFLG_TIF_MASK;   

    // Read current waveform sample 
    uint16_t v = table[idx];  

    DAC0->DAT[0].DATL = (v & 0xFF);
    DAC0->DAT[0].DATH = ((v >> 8) & 0x0F);

    // Move to the next sample
    idx++;

    // Start back at the 0th sample once you reach the 200th index
    if (idx >= FF){
			idx = 0;
		}
    LED_Off();  
}

int main(void){

    MCG_Clock120_Init();
    Switches_Init();
    LED_Init();
    DAC_Init();
    TimerInt_Init(); 

    __disable_irq();
    table = getTable(curK);  // Grab initialized waveform table at start up (curK=1)
    idx = 0;                 // Start at the 0th sample
    __enable_irq();

    while (1)
    {
        // Increments K on user input
        if (SW2Pressed())
        {
            Debounce();

            if (SW2Pressed())
            {
                if (curK < KMAX) curK++;   // Limit to max allowed K (10)

                __disable_irq();
                table = getTable(curK);   // Switch waveform 
                idx = 0;                  // Restart waveform cleanly
                __enable_irq();

                while (SW2Pressed()) { }  // Wait for release
                Debounce();
            }
        }

        // Decrements K on user input
        if (SW3Pressed())
        {
            Debounce();

            if (SW3Pressed())
            {
                if (curK > 1) curK--;     // Limit to minimum K (1)

                __disable_irq();
                table = getTable(curK);
                idx = 0;
                __enable_irq();

                while (SW3Pressed()) { }
                Debounce();
            }
        }
    }
}
